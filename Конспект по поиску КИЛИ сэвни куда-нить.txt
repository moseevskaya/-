. -- любой смв
.+ -- любой символ 1 и больше раз
.* -- любой символ 0 и больше раз
	пар.+а -- найдётся парта, парочка, пармезанка и прочие
	пар.*а -- ко всему прочему найдётся пара

[] -- задаёт выбор между несколькими эл-ми
	s[е л ь]d -- sеd, sлd, sьd 
	ва[к-н п-т]ся 
	[а-я ё\-]+ -- найдёт все русские слова
	[^_.,;:!?]+ -- любые символы, кроме заданных
	
\r, \n -- начало и конец строки
	^.*\r\n -- строка 
	^.*$ -- то же самое

Дано: <p> Ляляля </p>
	<.+> найдёт <, p, > и удалит к феням всё, что между ними. Но там может быть К текста. Так что жадный поиск идёт далеко
	<.*?> -- найдёт только сами теги кусками
	
^\r<.+?>.*?
